backwdSbst.m

                function [ x ] = backwdSbst ( U, b, n )

                for j=n:-1:1       

                               if U(j,j)==0

                                               break;

                               end   

                               x(j)=b(j)/U(j,j);   

                               for i=1:(j-1)

                                               b(i)=b(i)-U(i,j)*x(j);

                               end   

                end

                x=x';

 

fwdSbst.m

            function [ x ] = fwdSbst ( L, b, n )

                for j=1:n   

                               if L(j,j)==0

                                               break;

                               end   

                               x(j)=b(j)/L(j,j);   

                               for i=(j+1):n

                                               b(i)=b(i)-L(i,j)*x(j);

                               end   

                end

                x=x';

 

gaussElm.m

            function [ A, M ] = gaussElm ( A, n )

                for k=1:(n-1)   

                               if A(k,k)==0

                                               break;

                               end   

                               for i=(k+1):n

                                               M(i,k)=A(i,k)/A(k,k);

                               end    

                               for j=(k+1):n       

                                               for i=(k+1):n

                                                               A(i,j)=A(i,j)-M(i,k)*A(k,j);

                                               end       

                               end

                end

 

                for i=1:n

                               M(i,i)=1;

                end

 

                for i=1:n   

                               for j=1:n

                                               if i==j

                                                               break;

                                               else

                                                               A(i,j)=0;

                                               end

                               end   

                end



gram_schmidt.m

                function [ Q, R ] = gram_schmidt ( A, m, n )

                for k=1:n   

                               Q(:,k)=A(:,k);   

                               for j=1:(k-1)       

                                               R(j,k)=Q(:,j)'*A(:,k);

                                               Q(:,k)=Q(:,k)-R(j,k)*Q(:,j);

                               end   

                               Qk2=0;

                               for i=1:m

                                               Qk2=Qk2+Q(i,k)*Q(i,k);

                               end   

                               Qk2=sqrt(Qk2);       

                               R(k,k)=Qk2;   

                               if R(k,k)==0

                                                break;

                               end   

                               Q(:,k)=Q(:,k)/R(k,k);

                end
